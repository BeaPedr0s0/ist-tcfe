\begin{equation}
	v_s(t) = V_s u(-t) + sin(2 \pi f t)u(t)
	\label {equation:n1}
\end{equation}
where
\begin{equation}
	u(t)=e
	\begin{cases}
		0 & $t$ < $0$ \\
		1 & $t $\geq$ 0$
	\end{cases}
	\label {eq:n2}
\end{equation}
